Simple Example

// Define your filterable properties
public enum UserPropertyRef implements PropertyReference {
    NAME(String.class, Set.of(Op.EQ, Op.MATCHES)),
    AGE(Integer.class, Set.of(Op.EQ, Op.GT, Op.GTE, Op.LT, Op.LTE, Op.RANGE)),
    STATUS(UserStatus.class, Set.of(Op.EQ, Op.NE, Op.IN));
}

// Build dynamic filters with type safety
FilterRequest<UserPropertyRef> request = FilterRequest.<UserPropertyRef>builder()
    .filter("adults", new FilterDefinition<>(UserPropertyRef.AGE, Op.GTE, 18))
    .filter("active", new FilterDefinition<>(UserPropertyRef.STATUS, Op.EQ, UserStatus.ACTIVE))
    .filter("namePattern", new FilterDefinition<>(UserPropertyRef.NAME, Op.MATCHES, "John%"))
    .combineWith("(adults & active) | namePattern")
    .build();

// Execute with Spring Data JPA
FilterContext<User, UserPropertyRef> context = new FilterContext<>(User.class, UserPropertyRef.class);
Specification<User> spec = context.buildSpecification(request);
List<User> users = userRepository.findAll(spec);
